class Person{
    String? name; // optional variable
    int age;

    Person({@required String name,int age=20}){// named arguments
    // here 20 is default value
    // @required noted that is mandatory
    // @required not built in dart it is flutter feature
        this.name=name;
        this.age=age;
    }
}
